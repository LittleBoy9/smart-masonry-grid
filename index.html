App.tsx 
import { Masonry } from 'smart-masonry-grid/react';

function Gallery({ photos }) {
  return (
    <Masonry
      columns={{ sm: 2, md: 3, lg: 4 }}
      gap={16}
      animate
    >
      {photos.map((photo) => (
        <img
          key={photo.id}
          src={photo.src}
          alt={photo.alt}
        />
      ))}
    </Masonry>
  );
}
